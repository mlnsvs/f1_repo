sns.set(font='Helvetica')
sns.set_style("ticks")

for column in f1:
    # print(column)
    if (column == "team"):
        continue
    else:
        sns.barplot(y=f1["team"], x=f1[column], order=f1.sort_values(column, ascending=False).team, color = 'g')
        plt.title("Number of " + column.title())
        plt.xlabel(" ")
        plt.ylabel("Team")
        sns.despine()
        plt.show()
        plt.clf()
